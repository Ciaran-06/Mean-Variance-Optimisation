cmake_minimum_required(VERSION 3.14)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
project(MeanVarianceOpt)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# CPR (installed system-wide earlier)
find_package(cpr REQUIRED)

# Eigen (installed via Homebrew or system)
find_package(Eigen3 REQUIRED)

# nlohmann/json (header-only, can use built-in CMake support)
find_package(nlohmann_json REQUIRED)

add_executable(main main.cpp)

# Link libraries and include directories
target_link_libraries(main PRIVATE cpr::cpr nlohmann_json::nlohmann_json)
target_include_directories(main PRIVATE ${EIGEN3_INCLUDE_DIR})
